import pandas as pd

# Load both datasets
all_star = pd.read_excel('all_star.xlsx')
fielding = pd.read_csv('fielding.csv')

# Create all-star flag (1 if player participated in all-star game)
all_star['all_star_flag'] = 1

# Merge fielding with all_star data
merged = pd.merge(
    fielding,
    all_star[['player_id', 'year', 'all_star_flag']].drop_duplicates(),
    on=['player_id', 'year'],
    how='left'
)

# Fill NA for all_star_flag (0 = not all-star)
merged['all_star_flag'] = merged['all_star_flag'].fillna(0)

# Filter to years present in both datasets (1933-2015)
merged = merged[merged['year'] >= 1933]

# Clean position data - combine outfield positions
merged['pos_group'] = merged['pos'].replace({
    'LF': 'OF', 
    'RF': 'OF', 
    'CF': 'OF'
})

# Select relevant columns and drop rows missing key fielding metrics
analysis_df = merged[['player_id', 'year', 'pos', 'pos_group', 'all_star_flag', 
                     'po', 'a', 'e', 'zr', 'inn_outs']].dropna(
    subset=['po', 'a', 'e'])

# Verify merge results
print("Merged Data Information:")
print("=======================")
print(f"Shape: {analysis_df.shape}")
print("\nAll-star distribution:")
print(analysis_df['all_star_flag'].value_counts(normalize=True))

import pandas as pd
import numpy as np
from scipy.stats import mannwhitneyu
from statsmodels.stats.multitest import multipletests

# Load cleaned data
df = pd.read_csv('cleaned_fielding_all_star.csv')

# 1. Mann-Whitney U Tests by Position and Metric
positions = df['pos_group'].unique()
metrics = ['po', 'a', 'e']
results = []

for pos in positions:
    pos_data = df[df['pos_group'] == pos]
    for metric in metrics:
        # Split data into all-star and non-all-star
        all_star = pos_data[pos_data['all_star_flag'] == 1][metric]
        non_all_star = pos_data[pos_data['all_star_flag'] == 0][metric]
        
        # Perform test
        stat, p = mannwhitneyu(all_star, non_all_star, alternative='two-sided')
        
        # Calculate effect size (rank-biserial correlation)
        n1, n2 = len(all_star), len(non_all_star)
        r = 1 - (2 * stat) / (n1 * n2)
        
        results.append({
            'position': pos,
            'metric': metric,
            'statistic': stat,
            'p_value': p,
            'effect_size': r,
            'all_star_mean': all_star.mean(),
            'non_all_star_mean': non_all_star.mean()
        })

# Create results dataframe and adjust p-values
results_df = pd.DataFrame(results)
results_df['p_adj'] = multipletests(results_df['p_value'], method='fdr_bh')[1]

# Save results
results_df.to_csv('mann_whitney_results.csv', index=False)
print("Mann-Whitney test results saved to 'mann_whitney_results.csv'")

# 2. Correlation Analysis
# Select numeric features and create position dummies

print("Correlation heatmap saved to 'fielding_metrics_correlation.png'")
