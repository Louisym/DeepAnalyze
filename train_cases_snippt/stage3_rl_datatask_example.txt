import pandas as pd
import json

# Load employees data properly
with open('employees.json', 'r') as f:
    employees = [json.loads(line) for line in f]
    
employees_df = pd.DataFrame(employees)

# Load customers data with proper encoding
customers_df = pd.read_csv('customers.csv', encoding='latin1')

# Find Steve Johnson
steve_johnson = employees_df[
    (employees_df['first_name'].str.contains('Steve|Steven', case=False, regex=True)) &
    (employees_df['last_name'].str.contains('Johnson', case=False))
]

if len(steve_johnson) == 0:
    print("Steve Johnson not found in employees")
    print("\nAll employees:")
    print(employees_df[['id', 'first_name', 'last_name', 'title']])
else:
    steve_id = steve_johnson.iloc[0]['id']
    print(f"Found Steve Johnson with ID: {steve_id}")
    
    # Verify support_rep_id references
    invalid_support_ids = set(customers_df['support_rep_id']) - set(employees_df['id'])
    if invalid_support_ids:
        print(f"\nWarning: {len(invalid_support_ids)} invalid support_rep_id values")
    else:
        print("\nAll support_rep_id values are valid")
    
    # Count customers
    customer_count = len(customers_df[customers_df['support_rep_id'] == steve_id])
    print(f"\nSteve Johnson supports {customer_count} customers")
    
    # Show sample of supported customers
    supported_customers = customers_df[customers_df['support_rep_id'] == steve_id]
    print("\nSample of supported customers:")
    print(supported_customers[['first_name', 'last_name', 'email']].head())
